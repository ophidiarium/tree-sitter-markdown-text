        level: (atx_h2_marker)
        heading_content: (inline))
      (section
        (atx_heading
          level: (atx_h3_marker)
          heading_content: (inline))
        (section
          (atx_heading
            level: (atx_h4_marker)
            heading_content: (inline))
          (section
            (atx_heading
              level: (atx_h5_marker)
              heading_content: (inline))
            (section
              (atx_heading
                level: (atx_h6_marker)
                heading_content: (inline))
              (paragraph
                (inline)))))))))
